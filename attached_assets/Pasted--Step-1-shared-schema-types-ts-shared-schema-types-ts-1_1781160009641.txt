Step 2. server/services/evaluationEngine.ts 수정

파일:

server/services/evaluationEngine.ts

현재 analyzeBargeIn()은 이미 있습니다.
이 함수를 아래처럼 확장하는 방식이 좋습니다.

교체 권장 코드
export function analyzeBargeIn(messages: ConversationMessage[]): {
  count: number;
  positiveCount: number;
  negativeCount: number;
  neutralCount: number;
  rate: number;
  contexts: Array<{
    aiMessage: string;
    userMessage: string;
    assessment: 'positive' | 'negative' | 'neutral';
    reason: string;
    severity: 0 | 1 | 2 | 3 | 4;
  }>;
  netScoreAdjustment: number;
  summary: string;
} {
  const contexts: Array<{
    aiMessage: string;
    userMessage: string;
    assessment: 'positive' | 'negative' | 'neutral';
    reason: string;
    severity: 0 | 1 | 2 | 3 | 4;
  }> = [];

  let positiveCount = 0;
  let negativeCount = 0;
  let neutralCount = 0;

  const aiTurnCount = messages.filter(m => m.sender === 'ai').length || 1;

  messages.forEach((msg, idx) => {
    if (msg.sender === 'ai' && msg.interrupted) {
      const nextUserMsg = messages[idx + 1];

      if (!nextUserMsg || nextUserMsg.sender !== 'user') return;

      const aiText = msg.message || '';
      const userText = nextUserMsg.message || '';
      const normalizedUserText = userText.trim();

      let assessment: 'positive' | 'negative' | 'neutral' = 'neutral';
      let reason = '짧은 반응 또는 맥락상 영향이 크지 않은 중간 개입입니다.';
      let severity: 0 | 1 | 2 | 3 | 4 = 1;

      const aiLooksLikeQuestion =
        aiText.includes('?') ||
        /어떻|무엇|왜|어디|누가|언제|how|what|why|where|who|when/i.test(aiText);

      const userLooksLikeClarification =
        /제가 이해한|정리하면|확인하면|말씀하신|오해|정정|잠깐|먼저|핵심은|요지는/.test(normalizedUserText);

      const userLooksDefensive =
        /아니|그건 아니|이미 말했|제 말은|그게 아니라|왜냐하면|변명|문제가 아니/.test(normalizedUserText);

      const userHasSubstance = normalizedUserText.length >= 30;

      if (userLooksLikeClarification && userHasSubstance) {
        assessment = 'positive';
        reason = '상대 발화를 끊었지만, 내용을 정리·확인·정정하며 대화 흐름을 구조화한 개입입니다.';
        severity = 0;
        positiveCount++;
      } else if (aiLooksLikeQuestion && !userLooksLikeClarification) {
        assessment = 'negative';
        reason = 'AI가 질문 또는 우려를 제시하는 도중 개입하여 경청 부족으로 인식될 수 있습니다.';
        severity = userLooksDefensive ? 3 : 2;
        negativeCount++;
      } else if (userHasSubstance && !userLooksDefensive) {
        assessment = 'positive';
        reason = '중간 개입이 있었지만, 실질적인 정보 제공 또는 대화 주도에 기여했습니다.';
        severity = 1;
        positiveCount++;
      } else {
        assessment = 'neutral';
        reason = '짧은 응답 또는 맞장구 수준으로 대화 품질에 큰 영향은 제한적입니다.';
        severity = 1;
        neutralCount++;
      }

      contexts.push({
        aiMessage: aiText.substring(0, 120) + (aiText.length > 120 ? '...' : ''),
        userMessage: userText.substring(0, 120) + (userText.length > 120 ? '...' : ''),
        assessment,
        reason,
        severity,
      });
    }
  });

  const netScoreAdjustment =
    positiveCount * BARGE_IN_POSITIVE_BONUS - negativeCount * BARGE_IN_NEGATIVE_PENALTY;

  const cappedAdjustment = Math.max(
    BARGE_IN_MIN_ADJUSTMENT,
    Math.min(BARGE_IN_MAX_ADJUSTMENT, netScoreAdjustment)
  );

  const count = contexts.length;
  const rate = Math.round((count / aiTurnCount) * 100) / 100;

  const summary =
    count === 0
      ? '대화 중 AI 발화를 중단한 장면이 거의 없어 안정적인 경청 흐름을 보였습니다.'
      : negativeCount > positiveCount
        ? `총 ${count}회의 중간 개입 중 경청 부족으로 해석될 수 있는 장면이 ${negativeCount}회 확인되었습니다. 상대의 질문이나 우려가 끝난 뒤 요약하고 답변하는 연습이 필요합니다.`
        : positiveCount > 0
          ? `총 ${count}회의 중간 개입이 있었고, 이 중 ${positiveCount}회는 대화 정리·정보 제공·주도적 개입으로 볼 수 있습니다. 다만 반복될 경우 상대가 충분히 듣지 않는다고 느낄 수 있습니다.`
          : `총 ${count}회의 중간 개입이 있었으나 대부분 짧은 반응 수준으로, 대화 품질에 미치는 영향은 제한적입니다.`;

  return {
    count,
    positiveCount,
    negativeCount,
    neutralCount,
    rate,
    contexts,
    netScoreAdjustment: cappedAdjustment,
    summary,
  };
}

Step 6. 리포트 문구 기준

최종 리포트에는 아래 톤이 좋습니다.

나쁜 문구
말 끊기가 5회 발생했습니다. 경청 태도가 부족합니다.
좋은 문구
대화 중 중간 개입이 5회 있었습니다. 이 중 2회는 내용을 정리하거나 빠르게 정보를 제공한 적절한 개입이었지만, 3회는 상대방의 질문이나 우려가 끝나기 전에 답변하여 경청 부족으로 보일 수 있습니다. 다음 대화에서는 상대방의 말을 먼저 요약한 뒤 해결책을 제시하는 방식이 좋습니다.
